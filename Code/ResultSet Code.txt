    public connectivity1(ResultSet r) {
        try{
            rs=r;
            initComponents();
        mtdt=rs.getMetaData();
        for(i=1;i<=mtdt.getColumnCount();i++){
        mdl.addColumn(""+mtdt.getColumnName(i));
        }
        while(rs.next()){
          for(i=1;i<=mtdt.getColumnCount();i++)
          {al.add(""+rs.getObject(i));}
        mdl.addRow(al.toArray());
         al.clear();
        }
        
        res.setModel(mdl);
        }
        catch(SQLException s){}
    }
